 /*Table structure for table `faculty` */

DROP TABLE IF EXISTS `faculty`;

CREATE TABLE `faculty`(
`ID` BIGINT(20) NOT NULL,
`COLLEGE_ID` BIGINT(20) DEFAULT NULL,
`COLLEGE_NAME` VARCHAR(255) DEFAULT NULL,
`COURSE_ID`  BIGINT(20) DEFAULT NULL,
`COURSE_NAME` VARCHAR(255) DEFAULT NULL,
`FIRST_NAME` VARCHAR(255) DEFAULT NULL,
`LAST_NAME` VARCHAR(255) DEFAULT NULL,
`QUALIFICATION` VARCHAR(255) DEFAULT NULL,
`DATE_OF_BIRTH` DATETIME DEFAULT NULL,
`MOBILE_NO` VARCHAR(255) DEFAULT NULL,
`EMAIL` VARCHAR(255) DEFAULT NULL,
`ADDRESS` VARCHAR(255) DEFAULT NULL,
`CREATED_BY` VARCHAR(255) DEFAULT NULL,
`MODIFIED_BY` VARCHAR(255) DEFAULT NULL,
`CREATED_DATETIME` timestamp NULL default CURRENT_TIMESTAMP,
`MODIFIED_DATETIME` timestamp NULL default '0000-00-00 00:00:00',
PRIMARY KEY  (`ID`),
KEY `FIRST_LAST_NAME_IDX` (`FIRST_NAME`,`LAST_NAME`),
KEY `fk_faculty_IDX` (`COLLEGE_ID`),
KEY `fk_faculty2_IDX` (`COURSE_ID`),
 
CONSTRAINT `fk_faculty_IDX` FOREIGN KEY (`COLLEGE_ID`) REFERENCES `st_college` (`ID`) ON DELETE CASCADE ON UPDATE CASCADE ,
CONSTRAINT `fk_faculty2_IDX`  FOREIGN KEY  (`COURSE_ID`) REFERENCES  `course`  (`ID`) ON DELETE CASCADE ON UPDATE CASCADE
 
) ENGINE=INNODB DEFAULT CHARSET=latin1;
